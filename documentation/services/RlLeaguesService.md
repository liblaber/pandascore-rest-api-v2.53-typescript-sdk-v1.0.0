**Example Usage Code Snippet**

```typescript
import {
  FilterOverRlLeagues,
  Page,
  PandascoreClient,
  RangeOverRlLeagues,
  SearchOverRlLeagues,
} from 'pandascore_client';

(async () => {
  const pandascoreClient = new PandascoreClient({
    token: 'YOUR_TOKEN',
  });

  const filter: FilterOverRlLeagues = {
    id: [4],
    modifiedAt: ['la'],
    name: ['in ipsum en'],
    slug: ['wdih84'],
    url: ['laboris si'],
  };

  const range: RangeOverRlLeagues = {
    id: [1],
    modifiedAt: ['aute'],
    name: ['nostrud eiusmo'],
    slug: ['7aq-hl-w'],
    url: ['cupidatat occ'],
  };

  const search: SearchOverRlLeagues = {
    name: 'consectetur do',
    slug: 'g',
    url: 'eiusmod do',
  };
  const page = 1;

  const { data } = await pandascoreClient.rlLeagues.getRlLeagues({
    filter: filter,
    range: range,
    sort: [[]],
    search: search,
    page: page,
    perPage: 50,
  });

  console.log(data);
})();
```
