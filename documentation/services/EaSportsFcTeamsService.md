import {
  FilterOverFifaTeams,
  Page,
  PandascoreClient,
  RangeOverFifaTeams,
  SearchOverFifaTeams,
  VideogameId,
} from 'pandascore_client';

(async () => {
  const pandascoreClient = new PandascoreClient({
    token: 'YOUR_TOKEN',
  });

  const videogameId = VideogameId._1;

  const filter: FilterOverFifaTeams = {
    acronym: ['eiusmod nos'],
    id: [6],

    name: ['laborum aliqui'],
    slug: ['ev'],
  };

  const search: SearchOverFifaTeams = {
    acronym: 'Lorem nisi v',
    location: 'pariatur dolo',
    name: 'incidid',
    slug: 'm3zn',
  };
  const page = 1;

  const { data } = await pandascoreClient.eaSportsFcTeams.getFifaTeams({
    filter: filter,
    range: range,
    sort: [[]],
    search: search,
    page: page,
    perPage: 50,
  });

  console.log(data);
})();
