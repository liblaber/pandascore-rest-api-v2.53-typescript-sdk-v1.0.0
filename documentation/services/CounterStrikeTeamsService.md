```typescript
import {
  FilterOverCsgoTeams,
  Page,
  PandascoreClient,
  RangeOverCsgoTeams,
  SearchOverCsgoTeams,
  VideogameId,
} from 'pandascore_client';

(async () => {
  const pandascoreClient = new PandascoreClient({
    token: 'YOUR_TOKEN',
  });

  const videogameId = VideogameId._1;

  const filter: FilterOverCsgoTeams = {
    acronym: ['utinci'],
    id: [8],
    location: ['pariatur'],
    modifiedAt: ['es'],
    name: ['consectetur et '],
    slug: ['_j'],
    videogameId: [videogameId],
  };

  const range: RangeOverCsgoTeams = {
    acronym: ['veniam'],
    id: [6],
    location: ['dolore ut dolo'],
    modifiedAt: ['vel'],
    name: ['qui elit aliqua'],
    slug: ['x5z388t'],
  };

  const search: SearchOverCsgoTeams = {
    acronym: 'anim ',
    location: 'esse dolor ma',
    name: 'voluptate lab',
    slug: '3sr',
  };
  const page = 1;

  const { data } = await pandascoreClient.counterStrikeTeams.getCsgoTeams({
    filter: filter,
    range: range,
    sort: [[]],
    search: search,
    page: page,
    perPage: 50,
  });

  console.log(data);
})();
```
